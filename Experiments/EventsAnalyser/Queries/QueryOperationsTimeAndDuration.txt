from(bucket: "experiment_app")
  |> range(start: 0)
  |> filter(fn: (r) => r["_measurement"] == "spans")
  |> filter(fn: (r) => r["name"] == name )
  |> filter(fn: (r) => r["_field"] == "duration_nano" or (r["_field"] == "identity" and r["_value"] == identity))
  |> pivot(rowKey: ["_time", "span_id"], columnKey: ["_field"], valueColumn: "_value")
  |> filter(fn: (r) => r["identity"] != "")
  |> keep(columns: ["_time", "duration_nano"])
  |> map(fn: (r) => ({ r with _value: r.duration_nano}))
  |> sort(columns: ["_time"])
